// Simple email validation method
private boolean isValidEmail(String email) {
    if (email == null || email.isBlank()) return false;
    // Basic regex for email validation
    return email.matches("^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+$");
}
